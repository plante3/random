\newcount\Ca \newcount\Cb \newcount\Cc 

\def\gcd#1#2{%
    \begingroup
    \Ca#1\relax\Cb#2\relax
    \ifnum\Ca<\Cb \toendgroup{\gcd{#2}{#1}}\fi
    \Cc\Ca \divide\Cc\Cb \multiply\Cc-\Cb \advance\Ca\Cc
    \ifnum\Ca=\z@ \the\Cb \else 
        \Cc\Ca \gcd\Cb\Cc \fi
    \endgroup
}
\def\toendgroup#1#2\endgroup{\fi\endgroup#1}
