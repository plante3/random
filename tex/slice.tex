\protected\long\def\slice#1#2#3#4{%
    \begingroup \setbox0\hbox{#1}%
        \csname newbox\expandafter\endcsname \csname#2L\endcsname
        \csname newbox\expandafter\endcsname \csname#2C\endcsname
        \csname newbox\expandafter\endcsname \csname#2R\endcsname
        %
        \global\setbox\csname#2L\endcsname
            \hbox{\copy0 \kern\dimexpr#3\wd0-\wd0}%
        \global\setbox\csname#2C\endcsname
            \hbox{\kern-#3\wd0 \copy0 \kern\dimexpr#4\wd0-\wd0}%
        \global\setbox\csname#2R\endcsname
            \hbox{\kern-#4\wd0 \box0}%
        %
        \sliceA{#2L}\sliceA{#2C}\sliceA{#2R}
    \endgroup
}
\def\sliceA#1{%
    \pdfxform\csname#1\endcsname
        \global\setbox\csname#1\endcsname\hbox{\pdfrefxform\pdflastxform}%
}
\protected\def\sliceshow#1{%
    \hbox{\copy\csname#1L\endcsname\vrule width.2pt
    \copy\csname#1C\endcsname\vrule width.2pt
    \copy\csname#1R\endcsname}%
}
